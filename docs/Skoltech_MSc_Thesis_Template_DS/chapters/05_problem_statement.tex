\chapter{Problem statement}

\section{Preliminaries}

Let $\mathbf{x}_{\text{src}} \in \mathbb{R}^{3 \times H \times W}$ be a source image (possibly generated by the model from the source prompt), $P_{\text{src}}$ the source prompt, $P_{\text{tgt}}$ the target prompt, and $\psi$ a text encoder (CLIP for SD~v1.5, the built-in T5/SigLIP text encoder of FLUX.2 for the flow-matching backbone). A diffusion / flow model denoises over $T$ timesteps in a latent space $\mathbf{z}_t \in \mathbb{R}^{C \times h \times w}$.

For SD~v1.5 the DDIM~\cite{song2020denoising} step is
\begin{equation}
    \mathbf{z}_{t-1} = \sqrt{\bar{\alpha}_{t-1}}\,\hat{\mathbf{z}}_0(\mathbf{z}_t, c) + \sqrt{1 - \bar{\alpha}_{t-1}}\,\epsilon_\theta(\mathbf{z}_t, t, c),
    \label{eq::ddim_step}
\end{equation}
and classifier-free guidance~\cite{ho2022classifier}:
\begin{equation}
    \tilde{\epsilon}_\theta(\mathbf{z}_t, t, c) = \epsilon_\theta(\mathbf{z}_t, t, \emptyset_t) + s \cdot \left[\epsilon_\theta(\mathbf{z}_t, t, c) - \epsilon_\theta(\mathbf{z}_t, t, \emptyset_t)\right].
    \label{eq::cfg}
\end{equation}
For FLUX.2 the analogous step is a rectified-flow Euler update $\mathbf{z}_{t-1} = \mathbf{z}_t - \sigma_t \cdot v_\theta(\mathbf{z}_t, t, c)$ with CFG applied on top of the velocity $v_\theta$. From the perspective of DCS the two backbones are interchangeable: both take a per-step conditioning $c_t$ and produce $\mathbf{z}_{t-1}$.

\section{Discrete Context Switching}

\begin{Def}
\textbf{(Trajectory Mask.)} A binary vector $M = [m_T, \dots, m_1] \in \{0, 1\}^T$ selecting source or target conditioning per timestep.
\end{Def}

\begin{equation}
    c_t = m_t \cdot \psi(P_{\text{tgt}}) + (1 - m_t) \cdot \psi(P_{\text{src}}).
    \label{eq::context_switch}
\end{equation}

The hypothesis space has $|\mathcal{H}| = 2^T$ trajectories. Key subsets: pure source ($M = \mathbf{0}$), pure target ($M = \mathbf{1}$), forward step ($m_t = \mathbb{1}[t \leq \tau]$), reverse step ($m_t = \mathbb{1}[t > \tau]$), and stochastic ($m_t \sim \text{Bernoulli}(p)$). The all-target trajectory $M=\mathbf{1}$ is the default baseline: a user with no editing-specific machinery would run the target prompt through every timestep and get exactly this output.

\section{Mask Compression}

A mask has $B$ bits, each repeated $R = T_{\text{total}} / B$ times:
\begin{equation}
    \tilde{m}_t = m_{\lfloor t / R \rfloor}, \quad t = 1, \dots, T_{\text{total}}.
    \label{eq::mask_compression}
\end{equation}
For SD~v1.5 we use $T_{\text{total}} = 40$, $B = 20$, $R=2$: search space $2^{20} = 1{,}048{,}576$. For FLUX.2 we use $T_{\text{total}}=28$, $B = 14$, $R=2$: a strictly smaller search space that is still intractable to enumerate at 12~s per image.

\section{Segmentation-aware reward}
\label{sec:reward}

Let $\mathbf{B}\in\{0,1\}^{H\times W}$ be the background mask (1 = background, 0 = foreground), obtained by a segmentation model from a short \emph{seg prompt}. Given a generated image $\mathbf{x}_{\text{out}}=\text{Gen}(M)$, two scalar sub-rewards capture the two objectives:

\paragraph{Background preservation.}
\begin{equation}
f_{\text{bg}}(M) = \frac{\sum_{(i,j)} \mathbf{B}_{ij}\,\text{SSIM}(\mathbf{x}_{\text{src}}, \mathbf{x}_{\text{out}})_{ij}}{\sum_{(i,j)} \mathbf{B}_{ij}},
\end{equation}
a per-pixel SSIM~\cite{wang2004image} (Gaussian-windowed, $11\!\times\!11$, $\sigma=1.5$) averaged only over background pixels.

\paragraph{Foreground editability.}
The foreground is cropped from $\mathbf{x}_{\text{out}}$ using the bounding box of $1-\mathbf{B}$, padded to a square, and embedded with a vision encoder $\phi$ (CLIP for SD, SigLIP~2 for FLUX). The foreground score is
\begin{equation}
f_{\text{fg}}(M) = \cos\!\left(\phi(\text{crop}_{\text{fg}}(\mathbf{x}_{\text{out}})),\; \Delta_{\text{text}} / \|\Delta_{\text{text}}\|\right), \quad \Delta_{\text{text}} = \psi(P_{\text{tgt}}) - \psi(P_{\text{src}}).
\end{equation}
The delta direction isolates the semantic shift, cancelling shared context tokens that would otherwise dominate the similarity.

\paragraph{Combined reward for exhaustive search.}
In the exhaustive SD experiments both sub-rewards are min-max normalised across the $2^{20}$ samples and combined by geometric mean:
\begin{equation}
    \mathcal{S}_{\text{exh}}(M) = \sqrt{\hat{f}_{\text{bg}}(M) \cdot \hat{f}_{\text{fg}}(M)}, \qquad \hat{f}=(f - f_{\min})/(f_{\max}-f_{\min}).
    \label{eq::objective_exh}
\end{equation}
The geometric mean is zero if either term is zero, forcing both objectives to be met.

\paragraph{Combined reward for REINFORCE.}
Min-max normalisation is unavailable online, so we use a direct convex combination whose terms are already on a bounded scale:
\begin{equation}
    R(M) = f_{\text{bg}}(M)^{\alpha} \cdot \bigl(\text{relu}(f_{\text{fg}}(M)) + \varepsilon\bigr)^{1-\alpha},
    \label{eq::reward_reinforce}
\end{equation}
with $\alpha \in [0,1]$ trading background preservation off against edit strength and $\varepsilon=10^{-3}$ preventing the logarithm from diverging. $\alpha=0.7$ works best on background-rich scenes (the 15 overnight tasks of Section~\ref{sec:experiments_new_bgrich}); $\alpha=0.5$ is the safe default when the user has no prior.

\section{Optimization}

The editing problem is
\begin{equation}
    M^* = \arg\max_{M \in \{0,1\}^B} R(M),
    \label{eq::objective}
\end{equation}
and there are two regimes of interest:

\paragraph{Exhaustive (Chapter~\ref{ch:methods}, SD~v1.5).} Evaluate $R(M)$ for all $2^B$ masks; pick the argmax. Gives ground truth about the trajectory landscape but costs ${\sim}12$ GPU-hours per source image on $8\times$A100.

\paragraph{REINFORCE (Chapter~\ref{ch:methods}, FLUX.2).} Parameterise a Bernoulli policy $\pi_\theta(M)=\prod_{i=1}^{B}\text{Bern}(m_i\,|\,\sigma(\theta_i))$ and maximise $\mathbb{E}_{M\sim\pi_\theta}[R(M)]$ by REINFORCE with entropy regularisation:
\begin{equation}
\nabla_\theta J(\theta) \approx \mathbb{E}_{M\sim\pi_\theta}\bigl[(R(M)-b)\,\nabla_\theta \log \pi_\theta(M)\bigr] + \beta \nabla_\theta H(\pi_\theta),
\label{eq::reinforce_grad}
\end{equation}
where $b$ is an EMA baseline and $\beta$ is the entropy coefficient. The final mask is the greedy argmax $\hat{m}_i = \mathbb{1}[\sigma(\theta_i) > 0.5]$. This reduces the sampling budget from $2^B$ to a few hundred episodes.
